def delete_if
  if block_given?
    each do |value|
      delete(value) if yield(value)
    end
    self
  else
    nil
  end
end

def each
  if block_given?
    step_size = 100
    (0..length % step_size).step(step_size) do |step|
      first = step_size * step
      last = first + step_size
      list = connection.lrange id, first + 1, last
      list.each do |item|
        yield item
      end
    end
  else
    ::Enumerable::Enumerator.new(self, :each)
  end
end
